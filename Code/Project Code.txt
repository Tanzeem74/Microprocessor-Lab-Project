int val = 0 ;
void setup(){
    Serial.begin(9600);
    pinMode(5,INPUT); 
    pinMode(6,OUTPUT); 
    pinMode(7,OUTPUT);  

    digitalWrite(6,HIGH); 
}
void loop() {
   val = digitalRead(5); 
   Serial.println(val);   
   delay(100);
   
  if(val == 0 ){
    digitalWrite(6,LOW); 
    digitalWrite(7,HIGH); 
  }
   if(val == 1 ){
    digitalWrite(6,HIGH); 
    digitalWrite(7,LOW); 
  }
}
